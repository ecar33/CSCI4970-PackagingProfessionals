53 def _wait_for_file_ready(self, filepath, timeout=30):
54 """
55 @brief Wait for a file to be fully written by checking its size until it stabilizes or a timeout is reached. This helps ensure we don't try to process a PDF before it's fully saved.
56
57 @param filepath Path to the file to check
58 @param timeout Maximum time to wait in seconds before giving up
59 """
60 previous_size = -1
61 elapsed = 0
62 while elapsed < timeout:
63 try:
64 current_size = os.path.getsize(filepath)
65 except OSError:
66 time.sleep(1)
67 elapsed += 1
68 continue
69 if current_size == previous_size and current_size > 0:
70 return
71 previous_size = current_size
72 time.sleep(1)
73 elapsed += 1
