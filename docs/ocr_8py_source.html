7def extract_text_from_pdf(pdf_path):
8 """
9 @brief Extract text from a PDF file using OCR. This function converts each page of the PDF into an image and then uses Tesseract to extract text from those images.
10
11 @param pdf_path Path to the PDF file to process
12
13 @return A string containing the extracted text from the PDF
14
15 """
16 pages = convert_from_path(pdf_path, dpi=300)

24def parse_boxes_from_text(text):
25 """
26 @brief Parse box size and quantity information from the extracted text. This function looks for lines that contain box dimension patterns (like "18X18X18" or "12x10x6") and extracts the box size and the quantity received, which is assumed to be the last number on the line.
27
28 @param text The OCR-extracted text from which to parse box information
29
30 @return A list of dictionaries, each containing "box_size" (normalized to lowercase "DxDxD") and "count" (the quantity received)
31 """
32 boxes = []
33 # Match lines that contain a box dimension pattern like 18X18X18 or 12x10x6
34 box_pattern = re.compile(
35 r'(\d+)\s*[xX]\s*(\d+)\s*[xX]\s*(\d+)'
36 )
37
38 for line in text.split("\n"):
39 match = box_pattern.search(line)
40 if not match:
41 continue
42
43 # Normalize the box size to lowercase "DxDxD"
44 box_size = f"{match.group(1)}x{match.group(2)}x{match.group(3)}"
45
46 # The quantity received is the last number on the line
47 all_numbers = re.findall(r'\d+', line)
48 if not all_numbers:
49 continue
50 count = int(all_numbers[-1])
51
52 boxes.append({"box_size": box_size, "count": count})
53
54 return boxes
