36def increment_inventory_from_boxes(boxes):
37 """
38 @brief Increment inventory counts based on parsed box data from OCR results. This function takes a list of boxes (each with a size and count), finds the corresponding inventory item by matching the box size in the description, and increments the item quantity accordingly.
39
40 @param boxes A list of dictionaries, each containing "box_size" (e.g., "18x18x18") and "count" (quantity received) extracted from OCR parsing of order PDFs
41 """
42 with app.app_context():
43 for box in boxes:
44 box_size = box.get("box_size")
45 count = int(box.get("count", 0))
46 if not box_size or count <= 0:
47 continue
48
49 item = db.session.query(Inventory).filter(
50 Inventory.description.ilike(f"%{box_size}%")
51 ).first()
52
53 if item is None:
54 logger.warning(f"No inventory item found with '{box_size}' in description — skipping.")
55 continue
56
57 item.item_quantity += count
58 logger.info(f"Incremented SKU {item.sku} ({item.description}) by {count}")
59
60 db.session.commit()

62def decrement_inventory_from_sales(items):
63 """
64 @brief Decrement inventory counts based on sales data parsed from an uploaded CSV file. This function takes a list of items (each with SKU, description, sales count, and return count), finds the corresponding inventory item by SKU, and updates the item quantity by decrementing sales and incrementing returns.
65
66 @param items A list of dictionaries, each containing "sku", "description", "sales_count", and "return_count" parsed from the uploaded CSV file
67 """
68 for item_data in items:
69 sku = item_data.get("sku")
70 if not sku:
71 continue
72
73 sales_count = int(item_data.get("sales_count", 0))
74 return_count = int(item_data.get("return_count", 0))
75 item = db.session.get(Inventory, sku)
76
77 if item is None:
78 item = Inventory(
79 sku=sku,
80 description=item_data.get("description", sku),
81 item_quantity=max(0, return_count - sales_count),
82 return_quantity=return_count,
83 )
84 db.session.add(item)
85 continue
86
87 item.description = item_data.get("description", item.description)
88 item.item_quantity = max(0, item.item_quantity - sales_count + return_count)
89 item.return_quantity += return_count
90
91 db.session.commit()

This function accepts a JSON payload that can include "item_quantity" to set the quantity (must be a non-negative integer) and "description" to update the item's description (must not be empty). It validates the input, updates the corresponding inventory item in the database, and returns the updated item as JSON.
127
128 @param sku The SKU of the inventory item to update, provided as a URL parameter
129
130 @return A JSON representation of the updated inventory item, or an error message if the item is not found or if the input is invalid
131 """
132 item = db.session.get(Inventory, sku)
133 if item is None:
134 return jsonify(error="Inventory item not found"), 404
135
136 payload = request.get_json(silent=True) or {}
137
138 if "item_quantity" in payload:
139 try:
140 item.item_quantity = max(0, int(payload["item_quantity"]))

142 return jsonify(error="item_quantity must be an integer"), 400
143
144 if "description" in payload:

146 if not description:
147 return jsonify(error="description must not be empty"), 400
148 item.description = description
